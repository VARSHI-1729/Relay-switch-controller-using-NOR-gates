
module relay_switch(
    input A, B, C, D,
    output Y
);
    // Using 2-input NOR gates only
    wire nor1_out, nor2_out;
    
    // NOR1: NOR(C, C) = NOT C
    nor(nor1_out, C, C);
    
    // NOR2: NOR(A, NOT C)
    nor(nor2_out, A, nor1_out);
    
    // NOR3: NOR(B, NOR2_out) = Final output
    nor(Y, B, nor2_out);
    
endmodule
